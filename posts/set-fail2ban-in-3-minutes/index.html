Set Fail2ban in 3 Minutes

a simple protection on servers
January 17, 2025 
sudo apt install fail2ban # `sudo apt update`
sudo cp /etc/fail2ban/jail.conf /etc/fail2ban/jail.local
sudo vi /etc/fail2ban/jail.local # only edit local config file
[DEFAULT]
bantime = 10d # Ban duration: 10 days
findtime = 1d
maxretry = 3 # if 3 times fail in 1 day then ban
ignoreip = 127.0.0.1/8 ::1
[sshd]
enabled = true
port = ssh
filter = sshd
logpath = %(sshd_log)s # or `logpath = /var/log/auth.log`

sudo systemctl restart fail2ban
sudo systemctl enable fail2ban # Enable auto-start on boot
sudo systemctl status fail2ban.service

sudo fail2ban-client status sshd # Check SSH jail status
sudo tail -f /var/log/fail2ban.log # Monitor real-time bans
sudo iptables -L -n | grep REJECT
sudo cat /var/log/auth.log
  1. m-minute, d-day, none-second.
  2. search in vi: press /word then Enter. n for next and N for previous.
  3. ignoreip
    1. 127.0.0.1/8 refers to the IPv4 loopback address range. The /8 notation indicates this covers all addresses from 127.0.0.0 to 127.255.255.255. These addresses are used for local communications within the same machine.
    2. ::1 is the IPv6 loopback address, which serves the same purpose as 127.0.0.1 but for IPv6 networks.
    3. 多个 IP 地址之间使用空格分隔
  4. ref: Fail2Ban入门教程（非常详细） - C语言中文网
